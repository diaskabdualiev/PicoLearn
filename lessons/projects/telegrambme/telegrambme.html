        i2c = I2C(0, scl=Pin(I2C_SCL_PIN), sda=Pin(I2C_SDA_PIN))

        # –°–∫–∞–Ω–∏—Ä—É–µ–º —É—Å—Ç—Ä–æ–π—Å—Ç–≤–∞ –Ω–∞ —à–∏–Ω–µ I2C
        devices = i2c.scan()
        print(f"I2C —É—Å—Ç—Ä–æ–π—Å—Ç–≤–∞ –Ω–∞–π–¥–µ–Ω—ã –ø–æ –∞–¥—Ä–µ—Å–∞–º: {[hex(device) for device in devices]}")

        if BME280_ADDR not in devices:
            sensor_error_message = f"–î–∞—Ç—á–∏–∫ BME280 –Ω–µ –Ω–∞–π–¥–µ–Ω –ø–æ –∞–¥—Ä–µ—Å—É {hex(BME280_ADDR)}. –ü—Ä–æ–≤–µ—Ä—å—Ç–µ –ø–æ–¥–∫–ª—é—á–µ–Ω–∏–µ."
            print(sensor_error_message)
            return False

        # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è –¥–∞—Ç—á–∏–∫–∞ BME280
        bme = BME280(i2c=i2c, address=BME280_ADDR)

        # –ü—Ä–æ–≤–µ—Ä—è–µ–º, —Ä–∞–±–æ—Ç–∞–µ—Ç –ª–∏ –¥–∞—Ç—á–∏–∫, —Å—á–∏—Ç—ã–≤–∞—è –¥–∞–Ω–Ω—ã–µ
        values = bme.values
        print(f"–î–∞–Ω–Ω—ã–µ —Å –¥–∞—Ç—á–∏–∫–∞ BME280: {values}")

        sensor_initialized = True
        return True

    except Exception as e:
        sensor_error_message = f"–û—à–∏–±–∫–∞ –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏–∏ BME280: {str(e)}"
        print(sensor_error_message)
        return False

# –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è –¥–∞–Ω–Ω—ã—Ö —Å –¥–∞—Ç—á–∏–∫–∞
def read_sensor_data(force_update=False):
    global last_temp, last_pressure, last_humidity, last_update_time

    current_time = time.time()

    # –ü—Ä–æ–≤–µ—Ä—è–µ–º, –Ω—É–∂–Ω–æ –ª–∏ –æ–±–Ω–æ–≤–∏—Ç—å –¥–∞–Ω–Ω—ã–µ
    if not force_update and (current_time - last_update_time < UPDATE_INTERVAL):
        # –ò—Å–ø–æ–ª—å–∑—É–µ–º –∫—ç—à–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ
        return last_temp, last_pressure, last_humidity

    if not sensor_initialized:
        return None, None, None

    try:
        # –°—á–∏—Ç—ã–≤–∞–µ–º –¥–∞–Ω–Ω—ã–µ —Å –¥–∞—Ç—á–∏–∫–∞
        temp, pressure, humidity = bme.read_compensated_data()

        # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º –¥–∞–Ω–Ω—ã–µ –≤ —É–¥–æ–±–Ω—ã–π —Ñ–æ—Ä–º–∞—Ç
        temp = temp / 100.0  # –¢–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∞ –≤ –≥—Ä–∞–¥—É—Å–∞—Ö –¶–µ–ª—å—Å–∏—è
        pressure = pressure / 25600.0  # –î–∞–≤–ª–µ–Ω–∏–µ –≤ –≥–ü–∞
        humidity = humidity / 1024.0  # –í–ª–∞–∂–Ω–æ—Å—Ç—å –≤ –ø—Ä–æ—Ü–µ–Ω—Ç–∞—Ö

        # –û–±–Ω–æ–≤–ª—è–µ–º –∫—ç—à–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ
        last_temp = temp
        last_pressure = pressure
        last_humidity = humidity
        last_update_time = current_time

        # –ú–∏–≥–∞–µ–º —Å–≤–µ—Ç–æ–¥–∏–æ–¥–æ–º –ø—Ä–∏ –æ–±–Ω–æ–≤–ª–µ–Ω–∏–∏ –¥–∞–Ω–Ω—ã—Ö
        blink_led(1, 0.1)

        return temp, pressure, humidity

    except Exception as e:
        print(f"–û—à–∏–±–∫–∞ —á—Ç–µ–Ω–∏—è –¥–∞–Ω–Ω—ã—Ö —Å BME280: {e}")
        return None, None, None

# –§—É–Ω–∫—Ü–∏—è –¥–ª—è —Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏—è –¥–∞–Ω–Ω—ã—Ö –¥–ª—è –æ—Ç–ø—Ä–∞–≤–∫–∏ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—é
def format_sensor_data(temp, pressure, humidity):
    if temp is None or pressure is None or humidity is None:
        if sensor_error_message:
            return f"‚ùå –û—à–∏–±–∫–∞ –¥–∞—Ç—á–∏–∫–∞: {sensor_error_message}"
        return "‚ùå –î–∞–Ω–Ω—ã–µ —Å –¥–∞—Ç—á–∏–∫–∞ –Ω–µ–¥–æ—Å—Ç—É–ø–Ω—ã. –ü—Ä–æ–≤–µ—Ä—å—Ç–µ –ø–æ–¥–∫–ª—é—á–µ–Ω–∏–µ BME280."

    # –í—ã—á–∏—Å–ª—è–µ–º –≤—ã—Å–æ—Ç—É –Ω–∞–¥ —É—Ä–æ–≤–Ω–µ–º –º–æ—Ä—è –Ω–∞ –æ—Å–Ω–æ–≤–µ –¥–∞–≤–ª–µ–Ω–∏—è
    altitude = 44330 * (1 - (pressure/1013.25)**0.1903)  # –ü—Ä–∏–±–ª–∏–∑–∏—Ç–µ–ª—å–Ω–∞—è —Ñ–æ—Ä–º—É–ª–∞

    # –û–ø—Ä–µ–¥–µ–ª—è–µ–º –∫–æ–º—Ñ–æ—Ä—Ç–Ω–æ—Å—Ç—å —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä—ã
    temp_comfort = "ü•∂ —Ö–æ–ª–æ–¥–Ω–æ"
    if temp > 30:
        temp_comfort = "üî• –∂–∞—Ä–∫–æ"
    elif temp > 25:
        temp_comfort = "‚ô®Ô∏è —Ç–µ–ø–ª–æ"
    elif temp > 18:
        temp_comfort = "‚úÖ –∫–æ–º—Ñ–æ—Ä—Ç–Ω–æ"
    elif temp > 10:
        temp_comfort = "‚ùÑÔ∏è –ø—Ä–æ—Ö–ª–∞–¥–Ω–æ"

    # –û–ø—Ä–µ–¥–µ–ª—è–µ–º –∫–æ–º—Ñ–æ—Ä—Ç–Ω–æ—Å—Ç—å –≤–ª–∞–∂–Ω–æ—Å—Ç–∏
    humidity_comfort = "‚úÖ –Ω–æ—Ä–º–∞–ª—å–Ω–∞—è"
    if humidity > 70:
        humidity_comfort = "üíß –≤—ã—Å–æ–∫–∞—è"
    elif humidity < 30:
        humidity_comfort = "üèúÔ∏è –Ω–∏–∑–∫–∞—è"

    # –û–ø—Ä–µ–¥–µ–ª—è–µ–º —É—Å–ª–æ–≤–∏—è –¥–∞–≤–ª–µ–Ω–∏—è
    pressure_condition = "‚úÖ –Ω–æ—Ä–º–∞–ª—å–Ω–æ–µ"
    if pressure > 1020:
        pressure_condition = "‚òÄÔ∏è –≤—ã—Å–æ–∫–æ–µ"
    elif pressure < 1000:
        pressure_condition = "üåßÔ∏è –Ω–∏–∑–∫–æ–µ"

    # –§–æ—Ä–º–∏—Ä—É–µ–º —Å–æ–æ–±—â–µ–Ω–∏–µ —Å –¥–∞–Ω–Ω—ã–º–∏
    message = "üìä –î–∞–Ω–Ω—ã–µ —Å –¥–∞—Ç—á–∏–∫–∞ BME280:\n\n"
    message += f"üå°Ô∏è –¢–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∞: {temp:.1f}¬∞C ({temp_comfort})\n"
    message += f"üíß –í–ª–∞–∂–Ω–æ—Å—Ç—å: {humidity:.1f}% ({humidity_comfort})\n"
    message += f"üîÑ –î–∞–≤–ª–µ–Ω–∏–µ: {pressure:.1f} –≥–ü–∞ ({pressure_condition})\n"
    message += f"üèîÔ∏è –í—ã—Å–æ—Ç–∞: ~{altitude:.0f} –º –Ω–∞–¥ —É—Ä–æ–≤–Ω–µ–º –º–æ—Ä—è\n\n"

    # –î–æ–±–∞–≤–ª—è–µ–º –≤—Ä–µ–º—è –ø–æ—Å–ª–µ–¥–Ω–µ–≥–æ –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è
    last_update = time.localtime(last_update_time)
    message += f"üïí –î–∞–Ω–Ω—ã–µ –æ–±–Ω–æ–≤–ª–µ–Ω—ã: {last_update[3]:02d}:{last_update[4]:02d}:{last_update[5]:02d}"

    return message

def is_authorized(user_id):
    # –ï—Å–ª–∏ —Å–ø–∏—Å–æ–∫ –∞–≤—Ç–æ—Ä–∏–∑–æ–≤–∞–Ω–Ω—ã—Ö –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–π –ø—É—Å—Ç, —Ä–∞–∑—Ä–µ—à–∞–µ–º –≤—Å–µ–º
    if not AUTHORIZED_USERS:

    return user_id in AUTHORIZED_USERS

        else:
