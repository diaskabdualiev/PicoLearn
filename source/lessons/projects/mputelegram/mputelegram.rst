            time.sleep(0.05)
        
        # –í—ã—á–∏—Å–ª—è–µ–º —Å—Ä–µ–¥–Ω–∏–µ –∑–Ω–∞—á–µ–Ω–∏—è
        base_accel["x"] = sum_accel["x"] / samples
        base_accel["y"] = sum_accel["y"] / samples
        base_accel["z"] = sum_accel["z"] / samples
        
        base_angle["x"] = sum_angle["x"] / samples
        base_angle["y"] = sum_angle["y"] / samples
        
        print(f"–ë–∞–∑–æ–≤–æ–µ —É—Å–∫–æ—Ä–µ–Ω–∏–µ: X={base_accel['x']:.2f}g, Y={base_accel['y']:.2f}g, Z={base_accel['z']:.2f}g")
        print(f"–ë–∞–∑–æ–≤—ã–π —É–≥–æ–ª: X={base_angle['x']:.2f}¬∞, Y={base_angle['y']:.2f}¬∞")
        
        return True
    
    # –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø—Ä–æ–≤–µ—Ä–∫–∏ –≤–∏–±—Ä–∞—Ü–∏–∏ –∏ –Ω–∞–∫–ª–æ–Ω–∞
    def check_motion():
        global vibration_count, tilt_count
        
        # –°—á–∏—Ç—ã–≤–∞–µ–º —Ç–µ–∫—É—â–∏–µ –∑–Ω–∞—á–µ–Ω–∏—è
        accel_data = mpu.read_accel_data(True)  # –í –µ–¥–∏–Ω–∏—Ü–∞—Ö g
        gyro_data = mpu.read_gyro_data()  # –í –≥—Ä–∞–¥—É—Å–∞—Ö/—Å
        angle_data = mpu.read_angle()  # –í —Ä–∞–¥–∏–∞–Ω–∞—Ö
        
        # –ü—Ä–µ–æ–±—Ä–∞–∑—É–µ–º —Ä–∞–¥–∏–∞–Ω—ã –≤ –≥—Ä–∞–¥—É—Å—ã
        angle_deg = {
            "x": math.degrees(angle_data["x"]),
            "y": math.degrees(angle_data["y"])
        }
        
        # –í—ã—á–∏—Å–ª—è–µ–º —Ä–∞–∑–Ω–∏—Ü—É —É—Å–∫–æ—Ä–µ–Ω–∏—è –æ—Ç –±–∞–∑–æ–≤—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π
        accel_diff = {
            "x": abs(accel_data["x"] - base_accel["x"]),
            "y": abs(accel_data["y"] - base_accel["y"]),
            "z": abs(accel_data["z"] - base_accel["z"])
        }
        
        # –í—ã—á–∏—Å–ª—è–µ–º —Ä–∞–∑–Ω–∏—Ü—É —É–≥–ª–∞ –æ—Ç –±–∞–∑–æ–≤—ã—Ö –∑–Ω–∞—á–µ–Ω–∏–π
        angle_diff = {
            "x": abs(angle_deg["x"] - base_angle["x"]),
            "y": abs(angle_deg["y"] - base_angle["y"])
        }
        
        # –í—ã—á–∏—Å–ª—è–µ–º –º–∞–∫—Å–∏–º–∞–ª—å–Ω—É—é —Ä–∞–∑–Ω–∏—Ü—É —É—Å–∫–æ—Ä–µ–Ω–∏—è –ø–æ –≤—Å–µ–º –æ—Å—è–º
        max_accel_diff = max(accel_diff["x"], accel_diff["y"], accel_diff["z"])
        
        # –í—ã—á–∏—Å–ª—è–µ–º –º–∞–∫—Å–∏–º–∞–ª—å–Ω—É—é –∞–±—Å–æ–ª—é—Ç–Ω—É—é —É–≥–ª–æ–≤—É—é —Å–∫–æ—Ä–æ—Å—Ç—å –ø–æ –≤—Å–µ–º –æ—Å—è–º
        max_gyro = max(abs(gyro_data["x"]), abs(gyro_data["y"]), abs(gyro_data["z"]))
        
        # –í—ã—á–∏—Å–ª—è–µ–º –º–∞–∫—Å–∏–º–∞–ª—å–Ω—É—é —Ä–∞–∑–Ω–∏—Ü—É —É–≥–ª–∞ –ø–æ –æ—Å—è–º X –∏ Y
        max_angle_diff = max(angle_diff["x"], angle_diff["y"])
        
        # –û–ø—Ä–µ–¥–µ–ª—è–µ–º, –ø—Ä–µ–≤—ã—à–µ–Ω—ã –ª–∏ –ø–æ—Ä–æ–≥–∏
        vibration_detected = max_accel_diff > ACCEL_THRESHOLD or max_gyro > GYRO_THRESHOLD
        tilt_detected = max_angle_diff > ANGLE_THRESHOLD
        
        # –û–±–Ω–æ–≤–ª—è–µ–º —Å—á–µ—Ç—á–∏–∫–∏ —Å–æ–±—ã—Ç–∏–π
        if vibration_detected:
            vibration_count += 1
        
        if tilt_detected:
            tilt_count += 1
        
        return {
            "vibration": vibration_detected,
            "tilt": tilt_detected,
            "accel_diff": max_accel_diff,
            "gyro_max": max_gyro,
            "angle_diff": max_angle_diff
        }
