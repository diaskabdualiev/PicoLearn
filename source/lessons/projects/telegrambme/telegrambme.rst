        message += f"üíß –í–ª–∞–∂–Ω–æ—Å—Ç—å: {humidity:.1f}% ({humidity_comfort})\n"
        message += f"üîÑ –î–∞–≤–ª–µ–Ω–∏–µ: {pressure:.1f} –≥–ü–∞ ({pressure_condition})\n"
        message += f"üèîÔ∏è –í—ã—Å–æ—Ç–∞: ~{altitude:.0f} –º –Ω–∞–¥ —É—Ä–æ–≤–Ω–µ–º –º–æ—Ä—è\n\n"
        
        # –î–æ–±–∞–≤–ª—è–µ–º –≤—Ä–µ–º—è –ø–æ—Å–ª–µ–¥–Ω–µ–≥–æ –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è
        last_update = time.localtime(last_update_time)
        message += f"üïí –î–∞–Ω–Ω—ã–µ –æ–±–Ω–æ–≤–ª–µ–Ω—ã: {last_update[3]:02d}:{last_update[4]:02d}:{last_update[5]:02d}"
        
        return message
    
    # –§—É–Ω–∫—Ü–∏—è –¥–ª—è –ø—Ä–æ–≤–µ—Ä–∫–∏ –∞–≤—Ç–æ—Ä–∏–∑–∞—Ü–∏–∏ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
    def is_authorized(user_id):
        # –ï—Å–ª–∏ —Å–ø–∏—Å–æ–∫ –∞–≤—Ç–æ—Ä–∏–∑–æ–≤–∞–Ω–Ω—ã—Ö –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–π –ø—É—Å—Ç, —Ä–∞–∑—Ä–µ—à–∞–µ–º –≤—Å–µ–º
        if not AUTHORIZED_USERS:
            return True
        
        # –ò–Ω–∞—á–µ –ø—Ä–æ–≤–µ—Ä—è–µ–º, –µ—Å—Ç—å –ª–∏ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å –≤ —Å–ø–∏—Å–∫–µ
        return user_id in AUTHORIZED_USERS
    
    # –û–±—Ä–∞–±–æ—Ç—á–∏–∫ —Å–æ–æ–±—â–µ–Ω–∏–π –¥–ª—è –±–æ—Ç–∞
    def message_handler(bot, msg_type, chat_name, sender_name, chat_id, text, entry):
        print(f"–ü–æ–ª—É—á–µ–Ω–æ —Å–æ–æ–±—â–µ–Ω–∏–µ –æ—Ç {sender_name} ({chat_id}): {text}")
        
        # –ü–æ–ª—É—á–∞–µ–º ID –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è
        try:
            user_id = entry['message']['from']['id']
        except:
            user_id = None
        
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º –∞–≤—Ç–æ—Ä–∏–∑–∞—Ü–∏—é
        if not is_authorized(user_id):
            bot.send(chat_id, "‚ùå –£ –≤–∞—Å –Ω–µ—Ç –¥–æ—Å—Ç—É–ø–∞ –∫ –¥–∞–Ω–Ω—ã–º –¥–∞—Ç—á–∏–∫–∞.")
            return
        
        # –û–±—Ä–∞–±–æ—Ç–∫–∞ –∫–æ–º–∞–Ω–¥—ã /start
        if text == "/start":
            welcome_message = f"–ü—Ä–∏–≤–µ—Ç, {sender_name}! üëã\n"
            welcome_message += "–Ø –±–æ—Ç –¥–ª—è –º–æ–Ω–∏—Ç–æ—Ä–∏–Ω–≥–∞ –æ–∫—Ä—É–∂–∞—é—â–µ–π —Å—Ä–µ–¥—ã —Å –¥–∞—Ç—á–∏–∫–æ–º BME280.\n"
            welcome_message += "–û—Ç–ø—Ä–∞–≤—å –∫–æ–º–∞–Ω–¥—É /temp, —á—Ç–æ–±—ã –ø–æ–ª—É—á–∏—Ç—å —Ç–µ–∫—É—â–∏–µ –¥–∞–Ω–Ω—ã–µ –æ —Ç–µ–º–ø–µ—Ä–∞—Ç—É—Ä–µ, –≤–ª–∞–∂–Ω–æ—Å—Ç–∏ –∏ –¥–∞–≤–ª–µ–Ω–∏–∏."
            bot.send(chat_id, welcome_message)
        
        # –û–±—Ä–∞–±–æ—Ç–∫–∞ –∫–æ–º–∞–Ω–¥—ã /help
        elif text == "/help":
            help_message = "–î–æ—Å—Ç—É–ø–Ω—ã–µ –∫–æ–º–∞–Ω–¥—ã:\n"
            help_message += "/start - –ù–∞—á–∞—Ç—å —Ä–∞–±–æ—Ç—É —Å –±–æ—Ç–æ–º\n"
            help_message += "/help - –ü–æ–∫–∞–∑–∞—Ç—å —ç—Ç—É —Å–ø—Ä–∞–≤–∫—É\n"
            help_message += "/temp - –ü–æ–ª—É—á–∏—Ç—å —Ç–µ–∫—É—â–∏–µ –¥–∞–Ω–Ω—ã–µ —Å –¥–∞—Ç—á–∏–∫–∞\n"
            help_message += "/status - –ü–æ–∫–∞–∑–∞—Ç—å —Å—Ç–∞—Ç—É—Å —É—Å—Ç—Ä–æ–π—Å—Ç–≤–∞"
            bot.send(chat_id, help_message)
        
        # –û–±—Ä–∞–±–æ—Ç–∫–∞ –∫–æ–º–∞–Ω–¥—ã /status
        elif text == "/status":
            uptime_sec = time.time()
            uptime_min = uptime_sec // 60
            uptime_hours = uptime_min // 60
            
            status_message = "üìä –°—Ç–∞—Ç—É—Å —É—Å—Ç—Ä–æ–π—Å—Ç–≤–∞:\n"
            status_message += f"üïí –í—Ä–µ–º—è —Ä–∞–±–æ—Ç—ã: {int(uptime_hours)}—á {int(uptime_min % 60)}–º {int(uptime_sec % 60)}—Å\n"
            status_message += f"üíæ –°–≤–æ–±–æ–¥–Ω–∞—è –ø–∞–º—è—Ç—å: {gc.mem_free()} –±–∞–π—Ç\n"
            status_message += f"üå°Ô∏è –¢–µ–º–ø–µ—Ä–∞—Ç—É—Ä–∞ CPU: {machine.internal_temp() - 20} ¬∞C\n"  # –ö–æ—Ä—Ä–µ–∫—Ü–∏—è –¥–ª—è –±–æ–ª–µ–µ —Ç–æ—á–Ω—ã—Ö –ø–æ–∫–∞–∑–∞–Ω–∏–π
            
            if sensor_initialized:
                status_message += "üü¢ –î–∞—Ç—á–∏–∫ BME280: –ü–æ–¥–∫–ª—é—á–µ–Ω –∏ —Ä–∞–±–æ—Ç–∞–µ—Ç\n"
            else:
                status_message += f"üî¥ –î–∞—Ç—á–∏–∫ BME280: –û—à–∏–±–∫–∞ ({sensor_error_message})\n"
            
            last_update = time.localtime(last_update_time) if last_update_time > 0 else None
            if last_update:
                status_message += f"üîÑ –ü–æ—Å–ª–µ–¥–Ω–µ–µ –æ–±–Ω–æ–≤–ª–µ–Ω–∏–µ –¥–∞–Ω–Ω—ã—Ö: {last_update[3]:02d}:{last_update[4]:02d}:{last_update[5]:02d}"
            else:
                status_message += "üîÑ –î–∞–Ω–Ω—ã–µ —Å –¥–∞—Ç—á–∏–∫–∞ –µ—â–µ –Ω–µ —Å—á–∏—Ç—ã–≤–∞–ª–∏—Å—å"
            
            bot.send(chat_id, status_message)
        
        # –û–±—Ä–∞–±–æ—Ç–∫–∞ –∫–æ–º–∞–Ω–¥—ã /temp
        elif text == "/temp":
            # –°—á–∏—Ç—ã–≤–∞–µ–º –¥–∞–Ω–Ω—ã–µ —Å –¥–∞—Ç—á–∏–∫–∞ (—Å –ø—Ä–∏–Ω—É–¥–∏—Ç–µ–ª—å–Ω—ã–º –æ–±–Ω–æ–≤–ª–µ–Ω–∏–µ–º)
            temp, pressure, humidity = read_sensor_data(force_update=True)
            
            # –§–æ—Ä–º–∞—Ç–∏—Ä—É–µ–º –∏ –æ—Ç–ø—Ä–∞–≤–ª—è–µ–º —Å–æ–æ–±—â–µ–Ω–∏–µ
            message = format_sensor_data(temp, pressure, humidity)
            bot.send(chat_id, message)
        
        # –î–ª—è –Ω–µ–∏–∑–≤–µ—Å—Ç–Ω—ã—Ö –∫–æ–º–∞–Ω–¥
        else:
            bot.send(chat_id, "‚ùì –ù–µ–∏–∑–≤–µ—Å—Ç–Ω–∞—è –∫–æ–º–∞–Ω–¥–∞. –û—Ç–ø—Ä–∞–≤—å—Ç–µ /help –¥–ª—è —Å–ø–∏—Å–∫–∞ –¥–æ—Å—Ç—É–ø–Ω—ã—Ö –∫–æ–º–∞–Ω–¥.")
    
    # –ê—Å–∏–Ω—Ö—Ä–æ–Ω–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è –¥–ª—è –ø–µ—Ä–∏–æ–¥–∏—á–µ—Å–∫–æ–≥–æ —á—Ç–µ–Ω–∏—è –¥–∞–Ω–Ω—ã—Ö —Å –¥–∞—Ç—á–∏–∫–∞
    async def periodic_sensor_read():
        while True:
            try:
                # –°—á–∏—Ç—ã–≤–∞–µ–º –¥–∞–Ω–Ω—ã–µ –¥–ª—è –∫—ç—à–∏—Ä–æ–≤–∞–Ω–∏—è
                read_sensor_data(force_update=True)
                print(f"–ü–µ—Ä–∏–æ–¥–∏—á–µ—Å–∫–æ–µ –æ–±–Ω–æ–≤–ª–µ–Ω–∏–µ –¥–∞–Ω–Ω—ã—Ö: T={last_temp:.1f}¬∞C, P={last_pressure:.1f}–≥–ü–∞, H={last_humidity:.1f}%")
            except Exception as e:
                print(f"–û—à–∏–±–∫–∞ –ø—Ä–∏ –ø–µ—Ä–∏–æ–¥–∏—á–µ—Å–∫–æ–º —á—Ç–µ–Ω–∏–∏ –¥–∞—Ç—á–∏–∫–∞: {e}")
            
            # –ñ–¥–µ–º –¥–æ —Å–ª–µ–¥—É—é—â–µ–≥–æ –æ–±–Ω–æ–≤–ª–µ–Ω–∏—è
            await asyncio.sleep(UPDATE_INTERVAL)
    
    # –ê—Å–∏–Ω—Ö—Ä–æ–Ω–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è –¥–ª—è –∑–∞–ø—É—Å–∫–∞ –±–æ—Ç–∞
    async def run_bot():
        print("–ó–∞–ø—É—Å–∫ Telegram-–±–æ—Ç–∞...")
        bot = TelegramBot(BOT_TOKEN, message_handler)
        
        # –ó–∞–ø—É—Å–∫–∞–µ–º –æ—Å–Ω–æ–≤–Ω–æ–π —Ü–∏–∫–ª –±–æ—Ç–∞
        try:
            print("–ë–æ—Ç –∑–∞–ø—É—â–µ–Ω! –û–∂–∏–¥–∞–Ω–∏–µ –∫–æ–º–∞–Ω–¥...")
            await bot.run()
        except Exception as e:
            print(f"–û—à–∏–±–∫–∞ –≤ —Ä–∞–±–æ—Ç–µ –±–æ—Ç–∞: {e}")
            # –ú–∏–≥–∞–µ–º —Å–≤–µ—Ç–æ–¥–∏–æ–¥–æ–º –ø—Ä–∏ –æ—à–∏–±–∫–µ
            for _ in range(5):
                blink_led(3, 0.1)
                time.sleep(0.5)
    
    # –ì–ª–∞–≤–Ω–∞—è —Ñ—É–Ω–∫—Ü–∏—è –ø—Ä–æ–≥—Ä–∞–º–º—ã
    async def main():
        # –ò–º–ø–æ—Ä—Ç–∏—Ä—É–µ–º gc –¥–ª—è –æ—Ç—Å–ª–µ–∂–∏–≤–∞–Ω–∏—è –ø–∞–º—è—Ç–∏
        import gc
        gc.collect()
        
        print("–ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è...")
        
        # –ò–Ω–¥–∏–∫–∞—Ü–∏—è –∑–∞–ø—É—Å–∫–∞ —Å –ø–æ–º–æ—â—å—é —Å–≤–µ—Ç–æ–¥–∏–æ–¥–∞
        blink_led(3, 0.2)
        
        # –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º –¥–∞—Ç—á–∏–∫ BME280
        if init_bme280():
            print("–î–∞—Ç—á–∏–∫ BME280 —É—Å–ø–µ—à–Ω–æ –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä–æ–≤–∞–Ω!")
        else:
            print("–ù–µ —É–¥–∞–ª–æ—Å—å –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä–æ–≤–∞—Ç—å –¥–∞—Ç—á–∏–∫ BME280!")
        
        # –ü–æ–¥–∫–ª—é—á–∞–µ–º—Å—è –∫ Wi-Fi

        bot = TelegramBot(BOT_TOKEN, message_handler)
        bot.connect_wifi(WIFI_SSID, WIFI_PASSWORD)
        
        # –ú–∏–≥–∞–µ–º —Å–≤–µ—Ç–æ–¥–∏–æ–¥–æ–º –ø–æ—Å–ª–µ —É—Å–ø–µ—à–Ω–æ–≥–æ –ø–æ–¥–∫–ª—é—á–µ–Ω–∏—è
        blink_led(5, 0.1)
        
        print("Wi-Fi –ø–æ–¥–∫–ª—é—á–µ–Ω!")
        
        # –ó–∞–ø—É—Å–∫–∞–µ–º –±–æ—Ç–∞ –≤ –æ—Ç–¥–µ–ª—å–Ω–æ–π –∑–∞–¥–∞—á–µ
        bot_task = asyncio.create_task(run_bot())
        
        # –ó–∞–ø—É—Å–∫–∞–µ–º –ø–µ—Ä–∏–æ–¥–∏—á–µ—Å–∫–æ–µ —á—Ç–µ–Ω–∏–µ –¥–∞–Ω–Ω—ã—Ö –≤ –æ—Ç–¥–µ–ª—å–Ω–æ–π –∑–∞–¥–∞—á–µ
        sensor_task = asyncio.create_task(periodic_sensor_read())
        
        # –ó–∞–ø—É—Å–∫–∞–µ–º –ø–µ—Ä–∏–æ–¥–∏—á–µ—Å–∫—É—é –æ—á–∏—Å—Ç–∫—É –ø–∞–º—è—Ç–∏ –∏ –∏–Ω–¥–∏–∫–∞—Ü–∏—é —Ä–∞–±–æ—Ç—ã
        while True:
            gc.collect()
            led.on()  # –°–≤–µ—Ç–æ–¥–∏–æ–¥ –≥–æ—Ä–∏—Ç, –ø–æ–∫–∞–∑—ã–≤–∞—è, —á—Ç–æ –ø—Ä–æ–≥—Ä–∞–º–º–∞ —Ä–∞–±–æ—Ç–∞–µ—Ç
            await asyncio.sleep(2)
            led.off()
            await asyncio.sleep(2)
    
    # –ó–∞–ø—É—Å–∫–∞–µ–º –ø—Ä–æ–≥—Ä–∞–º–º—É —Å –ø–æ–º–æ—â—å—é –∞—Å–∏–Ω—Ö—Ä–æ–Ω–Ω–æ–≥–æ —Ü–∏–∫–ª–∞ —Å–æ–±—ã—Ç–∏–π
    if __name__ == "__main__":
        try:
            asyncio.run(main())
        except Exception as e:
            print(f"–ö—Ä–∏—Ç–∏—á–µ—Å–∫–∞—è –æ—à–∏–±–∫–∞: {e}")
            machine.reset()  # –ü–µ—Ä–µ–∑–∞–≥—Ä—É–∑–∫–∞ –ø—Ä–∏ –∫—Ä–∏—Ç–∏—á–µ—Å–∫–æ–π –æ—à–∏–±–∫–µ
